  • Action жҸҗдәӨзҡ„жҳҜ mutationпјҢиҖҢдёҚжҳҜзӣҙжҺҘеҸҳжӣҙзҠ¶жҖҒгҖӮ
  • Action еҸҜд»ҘеҢ…еҗ«д»»ж„ҸејӮжӯҘж“ҚдҪңгҖӮ

еҲҶеҸ‘action

Action еҮҪж•°жҺҘеҸ—дёҖдёӘдёҺ store е®һдҫӢе…·жңүзӣёеҗҢж–№жі•е’ҢеұһжҖ§зҡ„ context еҜ№иұЎпјҢеӣ жӯӨдҪ еҸҜд»Ҙи°ғз”Ё context.commit жҸҗдәӨдёҖдёӘ mutationпјҢжҲ–иҖ…йҖҡиҝҮ context.state е’Ң context.getters жқҘиҺ·еҸ– state е’Ң gettersгҖӮ

з»„д»¶дёӯеҸҜд»Ҙ this.$store.dispatch(вҖҳxxxвҖҷ) жқҘеҲҶеҸ‘

store/index.js

import Vue from 'vue'
import Vuex from 'vuex'
Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    count: 0
  },
  mutations: {
    add (state) {
      state.count++
    },
    addN (state, p) {
      state.count += p
    },
    addNObj (state, p) {
      console.log(p)
      state.count += p.n
    }
  },
  actions: {
    // addTimeout (context) {
    addTimeout ({ commit }) {
      // жү§иЎҢејӮжӯҘж“ҚдҪң
      setTimeout(() => {
        commit('add')
      }, 1000)
    },     
    addNTimeout ({ commit }, p) { //жҷ®йҖҡдј еҸӮ
      console.log(p)
      setTimeout(() => {// жү§иЎҢејӮжӯҘж“ҚдҪң
        commit('addN', p)
      }, 1000)
    },
    addNObjTimeout ({ commit }, p) {//еҜ№иұЎйЈҺж јдј еҸӮ
      console.log(p)
      setTimeout(() => {
        commit('addNObj', p)
      }, 1000)
    }
  }
})

store/index.js

import Vue from 'vue'
import Vuex from 'vuex'
Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    count: 0
  },
  mutations: {
    add (state) {
      state.count++
    },
    addN (state, p) {
      state.count += p
    }
  },
  actions: {
    // addTimeout (context) {
    addTimeout ({ commit }) {
      return new Promise((resolve, reject) => {
        // жү§иЎҢејӮжӯҘж“ҚдҪң
        setTimeout(() => {
          commit('add')
          resolve()
        }, 1000)
      })
    },
   /* addNTimeout ({ commit, dispatch }) {
      // е°ҶaddTimeoutиҝҷдёӘactionз»„еҗҲиҝӣжқҘ
      dispatch('addTimeout').then(() => {
        // жү§иЎҢејӮжӯҘж“ҚдҪң
        setTimeout(() => {
          commit('addN', 10)
        }, 1000)
      })
    }*/
      //з”Ёes6зҡ„async/awaitжқҘдјҳеҢ–
    async addNTimeout ({ commit, dispatch }) {
      // е°ҶaddTimeoutиҝҷдёӘactionз»„еҗҲиҝӣжқҘ
      await dispatch('addTimeout')
      // жү§иЎҢејӮжӯҘж“ҚдҪң
      console.log('1010101010')
      setTimeout(() => {
        commit('addN', 10)
      }, 1000)
    }
  }
})

modules

еҪ“еә”з”ЁеҸҳеҫ—йқһеёёеӨҚжқӮж—¶пјҢstore еҜ№иұЎе°ұжңүеҸҜиғҪеҸҳеҫ—зӣёеҪ“иҮғиӮҝгҖӮ

дёәдәҶи§ЈеҶід»ҘдёҠй—®йўҳпјҢVuex е…Ғи®ёжҲ‘д»¬е°Ҷ store еҲҶеүІжҲҗжЁЎеқ—пјҲmoduleпјүгҖӮжҜҸдёӘжЁЎеқ—жӢҘжңүиҮӘе·ұзҡ„ stateгҖҒmutationгҖҒactionгҖҒgetterгҖҒз”ҡиҮіжҳҜеөҢеҘ—еӯҗжЁЎеқ—вҖ”вҖ”д»ҺдёҠиҮідёӢиҝӣиЎҢеҗҢж ·ж–№ејҸзҡ„еҲҶеүІ

й»ҳи®Өжғ…еҶөдёӢпјҢжЁЎеқ—еҶ…йғЁзҡ„ actionгҖҒmutation е’Ң getter жҳҜжіЁеҶҢеңЁе…ЁеұҖе‘ҪеҗҚз©әй—ҙзҡ„вҖ”вҖ”иҝҷж ·дҪҝеҫ—еӨҡдёӘжЁЎеқ—иғҪеӨҹеҜ№еҗҢдёҖ mutation жҲ– action дҪңеҮәе“Қеә”гҖӮ

еҰӮжһңеёҢжңӣдҪ зҡ„жЁЎеқ—е…·жңүжӣҙй«ҳзҡ„е°ҒиЈ…еәҰе’ҢеӨҚз”ЁжҖ§пјҢдҪ еҸҜд»ҘйҖҡиҝҮж·»еҠ  namespaced: true зҡ„ж–№ејҸдҪҝе…¶жҲҗдёәеёҰе‘ҪеҗҚз©әй—ҙзҡ„жЁЎеқ—гҖӮ

жЁЎеқ—еҢ–иҜӯжі•жҖ»з»“пјҡ

з¬¬дёҖпјҡ $store.commitе’Ң$store.dispatch з¬¬дёҖдёӘеҸӮж•°дёәгҖҗжЁЎеқ—еҗҚ/ж–№жі•е‘ҪеҗҚгҖ‘
дҫӢеҰӮпјҡ
$store.commit('moduleA/addN',10)
$store.dispatch('moduleA/addNTimeout',10)

з¬¬дәҢпјҡ...mapXXX з¬¬дёҖдёӘеҸӮж•°дёәжЁЎеқ—еҗҚеӯ—з¬ҰдёІ,з¬¬дәҢдёӘеҸӮж•°е’Ңд»ҘеүҚзҡ„еҶҷжі•дёҖиҮҙ
дҫӢеҰӮпјҡ
...mapState('moduleA', ['count']),
...mapState('moduleA', { moduleAcount: 'count' }),

з¬¬дёү: 
...mapStateе’Ң...mapGettersеҶҷеңЁи®Ўз®—еұһжҖ§(computed)
...mapMutationsе’Ң...mapActionsеҶҷеңЁж–№жі•йҮҢ(methods)

гҖҗжЁЎеқ—еҢ–жЎҲдҫӢгҖ‘

store/index.js

import Vue from 'vue'
import Vuex from 'vuex'
Vue.use(Vuex)

// еҪ“з„¶д№ҹеҸҜд»ҘжҠҠmoduleAеҚ•зӢ¬ж”ҫе…ҘеҲ°дёҖдёӘjsж–Үд»¶дёӯ
const moduleA = {
  namespaced: true,//ејҖеҗҜе‘ҪеҗҚз©әй—ҙ
  state: {
    count: 123
  },
  getters: {
    getX2 (state) {
      return state.count * 2
    }
  },
  mutations: {
    addN (state, p) {
      state.count += p
    }
  },
  actions: {
    addNTimeout ({ commit }, p) {
      setTimeout(() => {
        commit('addN', p)
      }, 1000)
    }
  }
}
//moduleB ...
//modulec ...
export default new Vuex.Store({
  modules: {
    moduleA
      //moduleB ...
    //modulec ...
  }
})

<template>
  <div>
    <h1>жЁЎеқ—дёӯstateзҡ„дҪҝз”Ё</h1>
    <div>moduleAзҡ„countпјҡ{{ $store.state.moduleA.count }}</div>
    <div>moduleAзҡ„countпјҡ{{ count }}</div>
    <div>moduleAзҡ„countпјҡ{{ moduleAcount }}</div>
    <h1>жЁЎеқ—дёӯgetttersзҡ„дҪҝз”Ё</h1>
    <div>moduleAзҡ„gettersпјҡ{{ $store.getters['moduleA/getX2'] }}</div>
    <div>moduleAзҡ„gettersпјҡ{{ getX2 }}</div>
    <div>moduleAзҡ„gettersпјҡ{{ moduleAGetX2 }}</div>
     <h1>жЁЎеқ—дёӯmutationsзҡ„дҪҝз”Ё</h1>
    <div><button @click="$store.commit('moduleA/addN',10)">moduleAзҡ„mutations+10</button></div>
    <div><button @click="addN(10)">moduleAзҡ„mutations+10</button></div>
    <div><button @click="moduleAAddN(10)">moduleAзҡ„mutations+10</button></div>
     <h1>жЁЎеқ—дёӯactionsзҡ„дҪҝз”Ё</h1>
    <div><button @click="$store.dispatch('moduleA/addNTimeout',10)">moduleAзҡ„actions+10</button></div>
     <div><button @click="addNTimeout(10)">moduleAзҡ„actions+10</button></div>
     <div><button @click="moduleAAddNTimeout(10)">moduleAзҡ„actions+10</button></div>
  </div>
</template>

<script>
import { mapState, mapGetters, mapMutations, mapActions } from 'vuex'
export default {
  name: 'Count',
  computed: {
    ...mapState('moduleA', ['count']),
    ...mapState('moduleA', { moduleAcount: 'count' }),
    ...mapGetters('moduleA', ['getX2']),
    ...mapGetters('moduleA', { moduleAGetX2: 'getX2' })
  },
  methods: {
    ...mapMutations('moduleA', ['addN']),
    ...mapMutations('moduleA', { moduleAAddN: 'addN' }),
    ...mapActions('moduleA', ['addNTimeout']),
    ...mapActions('moduleA', { moduleAAddNTimeout: 'addNTimeout' })
  }
}
</script>
