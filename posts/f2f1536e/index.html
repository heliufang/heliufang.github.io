ж–ҮжЎЈең°еқҖпјҡhttp://nodejs.cn/api/fs.html

еңЁnodejsдёӯпјҢжҸҗдҫӣдәҶfsжЁЎеқ—пјҢиҝҷжҳҜnodeзҡ„ж ёеҝғжЁЎеқ—

жіЁж„Ҹпјҡ

  1. йҷӨдәҶglobalжЁЎеқ—дёӯзҡ„еҶ…е®№еҸҜд»ҘзӣҙжҺҘдҪҝз”ЁпјҢе…¶д»–жЁЎеқ—йғҪжҳҜйңҖиҰҒеҠ иҪҪзҡ„гҖӮ
  2. fsжЁЎеқ—дёҚжҳҜе…ЁеұҖзҡ„пјҢдёҚиғҪзӣҙжҺҘдҪҝз”ЁгҖӮеӣ жӯӨйңҖиҰҒеҜје…ҘжүҚиғҪдҪҝз”ЁгҖӮ
let fs = require("fs");

иҜӯжі•пјҡfs.readFile(path[,options], callback

path: еёҰж–Үд»¶еҗҚз§°зҡ„ж–Үд»¶и·Ҝеҫ„

optionsпјҡ ж–Үд»¶зј–з Ғ(еҸҜйҖү) иӢҘдёҚдј еҲҷиҝ”еӣһbufferеҜ№иұЎ

callback(err,data): ж–Үд»¶иҜ»е®Ңзҡ„еӣһи°ғ

ж–№ејҸ1пјҡдёҚдј зј–з ҒеҸӮж•°,еӣһи°ғеҮҪж•°дёӯеҸҜд»Ҙеҫ—еҲ°bufferеҜ№иұЎ,йңҖиҰҒдҪҝз”ЁtoStringиҪ¬еҢ–жҲҗеӯ—з¬ҰдёІ

    console.log(data) //жү“еҚ°bufferеҜ№иұЎ
    console.log(data.toString()) //жӯЈеёёжү“еҚ°еӯ—з¬ҰдёІ
})

let fs = require('fs')

е…ідәҺBufferеҜ№иұЎ

1. BufferеҜ№иұЎжҳҜNodejsз”ЁдәҺеӨ„зҗҶдәҢиҝӣеҲ¶ж•°жҚ®зҡ„гҖӮ
2. е…¶е®һд»»ж„Ҹзҡ„ж•°жҚ®еңЁи®Ўз®—жңәеә•еұӮйғҪжҳҜдәҢиҝӣеҲ¶ж•°жҚ®пјҢеӣ дёәи®Ўз®—жңәеҸӘи®ӨиҜҶдәҢиҝӣеҲ¶гҖӮ
3. жүҖд»ҘиҜ»еҸ–д»»ж„Ҹзҡ„ж–Үд»¶пјҢиҝ”еӣһзҡ„з»“жһңйғҪжҳҜдәҢиҝӣеҲ¶ж•°жҚ®пјҢеҚіBufferеҜ№иұЎ
4. BufferеҜ№иұЎеҸҜд»Ҙи°ғз”ЁtoString()ж–№жі•иҪ¬жҚўжҲҗеӯ—з¬ҰдёІгҖӮ

еҶҷж–Үд»¶

иҜӯжі•пјҡfs.writeFile(filepath, data[, options], callback)

filepathпјҡеёҰж–Үд»¶еҗҚз§°зҡ„ж–Үд»¶и·Ҝеҫ„

dataпјҡ иҰҒеҶҷе…Ҙзҡ„ж–Үд»¶еҶ…е®№

  2. requestдәӢд»¶жңүдёӨдёӘеҸӮж•°пјҢrequestиЎЁзӨәиҜ·жұӮеҜ№иұЎпјҢеҸҜд»ҘиҺ·еҸ–жүҖжңүдёҺиҜ·жұӮзӣёе…ізҡ„дҝЎжҒҜпјҢresponseжҳҜе“Қеә”еҜ№иұЎпјҢеҸҜд»ҘиҺ·еҸ–жүҖжңүдёҺе“Қеә”зӣёе…ізҡ„дҝЎжҒҜгҖӮ
  3. жңҚеҠЎеҷЁзӣ‘еҗ¬зҡ„з«ҜеҸЈиҢғеӣҙдёәпјҡ1-65535д№Ӣй—ҙпјҢжҺЁиҚҗдҪҝз”Ё3000д»ҘдёҠзҡ„з«ҜеҸЈпјҢеӣ дёә3000д»ҘдёӢзҡ„з«ҜеҸЈдёҖиҲ¬з•ҷз»ҷзі»з»ҹдҪҝз”Ё

CORS и·Ёеҹҹиө„жәҗе…ұдә«

  • CORSпјҲCross-Origin Resource SharingпјҢи·Ёеҹҹиө„жәҗе…ұдә«пјүи§ЈеҶіи·ЁеҹҹпјҢжҳҜйҖҡиҝҮ HTTP е“Қеә”еӨҙеҶіе®ҡжөҸи§ҲеҷЁжҳҜеҗҰйҳ»жӯўеүҚз«Ҝ JS д»Јз Ғи·ЁеҹҹиҺ·еҸ–иө„жәҗ
  • жөҸи§ҲеҷЁзҡ„еҗҢжәҗе®үе…Ёзӯ–з•Ҙй»ҳи®Өдјҡйҳ»жӯўзҪ‘йЎөвҖңи·ЁеҹҹвҖқиҺ·еҸ–иө„жәҗгҖӮдҪҶеҰӮжһңжҺҘеҸЈжңҚеҠЎеҷЁй…ҚзҪ®дәҶ CORS зӣёе…ізҡ„ HTTP е“Қеә”еӨҙпјҢе°ұеҸҜи§ЈйҷӨжөҸи§ҲеҷЁз«Ҝзҡ„и·Ёеҹҹи®ҝй—®йҷҗеҲ¶
  • CORS дё»иҰҒеңЁжңҚеҠЎеҷЁз«ҜиҝӣиЎҢй…ҚзҪ®гҖӮе®ўжҲ·з«ҜжөҸи§ҲеҷЁж— йЎ»еҒҡд»»дҪ•йўқеӨ–зҡ„й…ҚзҪ®пјҢеҚіеҸҜиҜ·жұӮејҖеҗҜдәҶ CORS зҡ„жҺҘеҸЈгҖӮ
  • CORS еңЁжөҸи§ҲеҷЁдёӯжңүе…је®№жҖ§гҖӮеҸӘжңүж”ҜжҢҒ XMLHttpRequest Level2 зҡ„жөҸи§ҲеҷЁпјҢжүҚиғҪжӯЈеёёи®ҝй—®ејҖеҗҜдәҶ CORS зҡ„жңҚеҠЎз«ҜжҺҘеҸЈпјҲдҫӢеҰӮпјҡIE10+гҖҒChrome4+гҖҒFireFox3.5+пјүгҖӮ

CORS еёёи§Ғе“Қеә”еӨҙ

  • Access-Control-Allow-OriginпјҡеҲ¶е®ҡдәҶе…Ғи®ёи®ҝй—®иө„жәҗзҡ„еӨ–еҹҹ URL

res.setHeader('Access-Control-Allow-Origin', '*')
  • Access-Control-Allow-Headers
  • й»ҳи®Өжғ…еҶөдёӢпјҢCORS д»…ж”ҜжҢҒе®ўжҲ·з«Ҝеҗ‘жңҚеҠЎеҷЁеҸ‘йҖҒеҰӮдёӢзҡ„ 9 дёӘиҜ·жұӮеӨҙпјҡAcceptгҖҒAccept-LanguageгҖҒContent-LanguageгҖҒDPRгҖҒDownlinkгҖҒSave-DataгҖҒViewport-WidthгҖҒWidth гҖҒContent-Type пјҲеҖјд»…йҷҗдәҺ text/plainгҖҒmultipart/form-dataгҖҒapplication/x-www-form-urlencoded дёүиҖ…д№ӢдёҖпјү
  • еҰӮжһңе®ўжҲ·з«Ҝеҗ‘жңҚеҠЎеҷЁеҸ‘йҖҒдәҶйўқеӨ–зҡ„иҜ·жұӮеӨҙдҝЎжҒҜпјҢеҲҷйңҖиҰҒеңЁжңҚеҠЎеҷЁз«ҜпјҢйҖҡиҝҮ Access-Control-Allow-Headers еҜ№йўқеӨ–зҡ„иҜ·жұӮеӨҙиҝӣиЎҢеЈ°жҳҺпјҢеҗҰеҲҷиҝҷж¬ЎиҜ·жұӮдјҡеӨұиҙҘпјҒ
res.setHeader('Access-Control-Allow-Headers', 'Content-Type, X-Custom-Header')
  • Access-Control-Allow-Methods
  • й»ҳи®Өжғ…еҶөдёӢпјҢCORS д»…ж”ҜжҢҒе®ўжҲ·з«ҜеҸ‘иө· GETгҖҒPOSTгҖҒHEAD иҜ·жұӮгҖӮеҰӮжһңе®ўжҲ·з«ҜеёҢжңӣйҖҡиҝҮ PUTгҖҒDELETE зӯүж–№ејҸиҜ·жұӮжңҚеҠЎеҷЁзҡ„иө„жәҗпјҢеҲҷйңҖиҰҒеңЁжңҚеҠЎеҷЁз«ҜпјҢйҖҡиҝҮ Access-Control-Alow-Methods жқҘжҢҮжҳҺе®һйҷ…иҜ·жұӮжүҖе…Ғи®ёдҪҝз”Ёзҡ„ HTTP ж–№жі•

  • Session и®ӨиҜҒжңәеҲ¶йңҖиҰҒй…ҚеҗҲ Cookie жүҚиғҪе®һзҺ°гҖӮз”ұдәҺ Cookie й»ҳи®ӨдёҚж”ҜжҢҒи·Ёеҹҹи®ҝй—®пјҢжүҖд»ҘпјҢеҪ“ж¶үеҸҠеҲ°еүҚз«Ҝи·ЁеҹҹиҜ·жұӮеҗҺз«ҜжҺҘеҸЈзҡ„ж—¶еҖҷпјҢйңҖиҰҒеҒҡеҫҲеӨҡйўқеӨ–зҡ„й…ҚзҪ®пјҢжүҚиғҪе®һзҺ°и·Ёеҹҹ Session и®ӨиҜҒгҖӮ
  • еҪ“еүҚз«ҜиҜ·жұӮеҗҺз«ҜжҺҘеҸЈдёҚеӯҳеңЁи·Ёеҹҹй—®йўҳзҡ„ж—¶еҖҷпјҢжҺЁиҚҗдҪҝз”Ё Session иә«д»Ҫи®ӨиҜҒжңәеҲ¶гҖӮ
  • еҪ“еүҚз«ҜйңҖиҰҒи·ЁеҹҹиҜ·жұӮеҗҺз«ҜжҺҘеҸЈзҡ„ж—¶еҖҷпјҢдёҚжҺЁиҚҗдҪҝз”Ё Session иә«д»Ҫи®ӨиҜҒжңәеҲ¶пјҢжҺЁиҚҗдҪҝз”Ё JWT и®ӨиҜҒжңәеҲ¶
